#자료형# 1) 문자형 (charactor): "사과", "Hello, World"# 2) 정수형(integer): 1324# 3) 논리형(logical): TRUE(1), FALSE(0)# 4) 복소수형(complex): 1+2i, 3+4i# 5) 실수형(double): 3.14# 6) 요인형(factor):Levels: A B C D# 자료형은 typeof(변수명) 로 확인가능 ex) typeof(a)a + b  # a와 b를 더함a - b # a에서 b를 뺌a * b  # a와 b를 곱함a / b  # a를 b로 나눔a %% b  # a를 b로 나눈 나머지a %/% b  # a를 b로 나눈 몫a^b  # a를 b번 곱함 (a의 b승)a %*% b  # a와 b가 같은 길이 벡터이면, a와 b의 내적 계산x<-4; n<-2abs(x)  # 절대값sqrt(x)  #제곱근ceiling(x)  # x 이상 최소 정수floor(x)  # x 이하 최대 정수trunc(x)  # 소수점 미만 버림round(x, digits=n)  # 소수점 아래 n자리까지 반올림signif(x, digits=n)  # 유효 숫자 n자리로 반올림cos(x)sin(x)tan(x)  # 삼각함수log(x)  # 자연 로그log10(x)  # 대수 로그exp(x)  # 지수함수 e^xmean(x, na.rm=FALSE)  # x의 평균sd(x, na.rm=FALSE)  # x의 표준편차median(x, na.rm=FALSE)  # x의 중앙값(중위수)quantile(x, na.rm=FALSE)  # x의 분위수#여러 변수 사용이 가능한 통계 함수 range(x, na.rm=FALSE)  # 전체의 최소치와 최대치sum(x, na.rm=FALSE)  # 전체의 합max(x, na.rm=FALSE)  # 전체의 최대치min(x, na.rm=FALSE)  # 전체의 최소치summary(x)  # 최소, (25, 50%), 평균, (75%), 최대치 var(x, na.rm=FALSE)  # x의 분산, 또는 분산-공분산 행렬scale(x, center=TRUE, scale=TRUE)  # x의 열 별 표준화 값# +++ 보충 자료 +++# 변수이름 규칙에 어긋나는 상황들2a <- 3   # 앞에 숫자가 있거나.2 <- c(9,6) # 특수문자가 있는 변수이름은 선언 불가# 평균 설정(mean_value 평균값 할당)mean_value <- 50# 서로 크기가 다른 sd (편차) 설정sd_small <- 5sd_large <- 200#  set.seed()를 사용하여 시드 값을 특정 숫자로 설정하면 코드를 실행할 때마다# 코드에서 생성된 난수가 동일하게 보장됩니다. # 시드값과 sd(표준편차)값을 여러 값으로 변경하여 실습해보시길 바랍니다.set.seed(124)# rnorm 을 사용하여 n 개의 값을 임의 추출하여 정규분포를 작성한다.# ex) rnorm(data = n, mean = , sd = )# 데이터 값, 평균, 표준편차값을 변경하여 여러 분포를 확인 할 수 있다.data_small_sd <- rnorm(1000, mean = mean_value, sd = sd_small)data_large_sd <- rnorm(1000, mean = mean_value, sd = sd_large)par(mfrow=c(2,2)) # Plots창(그림)을 2x2로 띄어준다# 평균은 동일하더라도 편차에 따라 분포의 정도가 다른걸 확인할 수 있다.hist(data_small_sd,  main = "Distribution with Small Standard Deviation", xlab = "Value")hist(data_large_sd, main = "Distribution with Large Standard Deviation", xlab = "Value")# Error in plot.new() : figure margins too large <- 해당 오류가 뜬 다면# 오른쪽 plots 늘려보시길 바랍니다.set.seed(1213)  # 시드값(난수)을 바꿔 실습해보시기 바랍니다 시드값은 자유 기입data_small_sd <- rnorm(1000, mean = mean_value, sd = sd_small)data_large_sd <- rnorm(1000, mean = mean_value, sd = sd_large)hist(data_small_sd,  main = "Distribution with Small Standard Deviation", xlab = "Value")hist(data_large_sd, main = "Distribution with Large Standard Deviation", xlab = "Value")
